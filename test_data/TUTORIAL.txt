########################################################################################################################################
#  USAGE :                                                                                                                              
#       python3 VPOT.py  -  will return a help screen                                                                                   
#       python3 VPOT.py priority <location for output file+prefix> <file of input VCF files>   will create a default parameter file   
#       python3 VPOT.py priority <location for output file+prefix> <file of input VCF files> <parameter file>                          
#       python3 VPOT.py genef <location for output file+prefix> <VPOT priority output> <gene list>                                  
#       python3 VPOT.py samplef <location for output file+prefix> <VPOT priority output> <sample selection file>                   
#       python3 VPOT.py stats <location for output file+prefix> <VPOT priority output> <a percentile value [1-99]>                   
#       python3 VPOT.py merge <location for output file+prefix> <file of VPOT priority output files>                    
#                                                                                                                                
########################################################################################################################################
# Prioritise your variants (VCF/TXT input)                                                                                           
# 
#          	python3 VPOT.py priority <location for output file+prefix> <file of input VCF/TXT files> <parameter file>                  
#                                                                                                                                       
# command :                                                                                                                             
#           change to test_data directory
#
# For VCF-	
#           python3 ../VPOT.py priority testout_ test_VCF_sample_list.txt test_variants_parameters.txt                                
#
#       	  files FAM_1.vcf, test_VCF_sample_list.txt and 
#                 test_variants_parameters.txt are supplied in the test_data directory of VPOT download  
#
# For TXT-	
#           python3 ../VPOT.py priority testout_ test_TXT_sample_list.txt test_variants_parameters.txt                           
#
#         	files FAM_1.txt and test_TXT_sample_list.txt are supplied in the test_data directory of VPOT download                      
#                                                                                                                                       
# result : an output file will be created                                                                                               
#          testout_final_output_file_XXXXXXXXXX.txt                                                                                     
#                                                                                                                                      
#######################################################################################################################################
#
#######################################################################################################################################
# Filter your output by gene list                                                                                                       
#
#       	python3 VPOT.py genef <location for output file+prefix> <VPOT priority output> <gene list>                                  
#                                                                                                                                    
# command :                                                                                                                        
#          	change to test_data directory
#
# 			    python3 ../VPOT.py genef testout_ testout_final_output_file_XXXXXXXXXX.txt test_gene.txt              
#
#         	file test_gene.txt is supplied in the test_data directory of VPOT download                                                   
#                                                                                                                                       
# result : an output file will be created                                                                                               
#          testout_gene_filtered_output_file_XXXXXXXXXX.txt                                                                             
#                                                                                                                                       
#######################################################################################################################################
#
#######################################################################################################################################
# Filter your output by sample pedigree                                                                                                 
#                                                                                                                                       
#       	python3 VPOT.py samplef <location for output file+prefix> <VPOT priority output> <sample selection file>                   
#
# command :                                                                                                                             
#       	change to test_data directory
#
# 			python3 ../VPOT.py samplef testout_ testout_final_output_file_XXXXXXXXXX.txt test_sample_set.ped                                 
#
#         	file test_sample_set.ped is supplied in the test_data directory of VPOT download                                             
#                                                                                                                                       
# result : an output file will be created                                                                                               
#          testout_variant_filtered_output_file_XXXXXXXX.txt                                                                             
#                                                                                                                                       
#######################################################################################################################################
#
#######################################################################################################################################
# Generate a summary statistics output file                                                                                        
#                                                                                                                                       
#       	python3 VPOT.py stats <location for output file+prefix> <VPOT priority output> <a percentile value [1-99] >              
#
# command :                                                                                                                             
#       	change to test_data directory
#
# 			python3 ../VPOT.py stats testout_ testout_final_output_file_XXXXXXXXXX.txt 99                                           
#
#				: for the input percentile value the summary provide a gene breakdown of variants that meet 
#				  the percentile criteria for each sample
#					- to find the highest priority variant/gene use a high value eg 90 for 90th percentile             
#					- to list all variants/genes for the sample use 1 for 1st percentile                             
#					- if no specific input value is given then a default value of 50 is used                             
#                               
# result : a statistics output file is create                                                                                           
#          testout_variant_statistic_file_XXXXXXXX.txt                                                                             
#                                                                                                                                       
#######################################################################################################################################
#
#######################################################################################################################################
# Merge multiple VPOT priority output files into one output file                                                                                        
#                                                                                                                                       
#       	python3 VPOT.py merge <location for output file+prefix> <file of VPOT priority output files>              
#
# command :                                                                                                                             
#       	change to test_data directory
#
# 			python3 ../VPOT.py merge testout_ test_merge_VPOL_list.txt      
#
#				: a new output file is created that will contain 
#							- all the unique variants from the input VPOL files 
#							- all the samples from the input VPOL files
#                               
# result : a consolidated output file merging variants and samples is create                                                       
#          testout_variant_merge_file_XXXXXXXX.txt                                                                             
#
